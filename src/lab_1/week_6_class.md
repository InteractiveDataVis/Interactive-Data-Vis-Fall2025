```js
let dream = 0; 
let biscoe = 0;
let torgersen = 0;
for (let p = 0; p < penguins.length; p++) {
  const penguin = penguins[p]
  if (penguin.island === "Dream") dream = dream + 1;
  if (penguin.island === "Biscoe") biscoe = biscoe + 1;
  if (penguin.island === "Torgersen") torgersen = torgersen + 1;
  // display(penguin)
}
```

 }
